// Système de Sécurité Domestique
#include 
#include 

#define PIR_PIN 2
#define BUZZER_PIN 8
#define VIBRATION_PIN 3

PIR pir(PIR_PIN);
Buzzer buzzer(BUZZER_PIN);

void setup() {
    Serial.begin(9600);
    pinMode(VIBRATION_PIN, INPUT);
}

void loop() {
    if(pir.detect()) {
        buzzer.on();
        Serial.println("ALERTE: Mouvement détecté!");
    }
    
    if(digitalRead(VIBRATION_PIN) == HIGH) {
        buzzer.on();
        Serial.println("ALERTE: Vibration détectée!");
    }
    
    delay(100);
}

Garage Intelligent

Simulation

Code Arduino

// Garage Intelligent
#include 
#include 

#define TRIG_PIN 5
#define ECHO_PIN 6
#define SERVO_PIN 9

Servo garageDoor;
UltraSonicDistanceSensor sonar(TRIG_PIN, ECHO_PIN);

void setup() {
    garageDoor.attach(SERVO_PIN);
    Serial.begin(9600);
}

void loop() {
    float distance = sonar.measureDistanceCm();
    
    if(distance < 50) {
        garageDoor.write(90); // Ouvrir
        Serial.println("Porte ouverte");
    }
    
    else {
        garageDoor.write(0); // Fermer
    }
    
    delay(500);
}

Voiture Télécommandée

Contrôleur

Direction: Neutre

Code Arduino

// Voiture Télécommandée
#include 

AF_DCMotor motor1(1);
AF_DCMotor motor2(2);

void setup() {
    Serial.begin(9600);
    motor1.setSpeed(255);
    motor2.setSpeed(255);
}

void loop() {
    if(Serial.available()) {
        char cmd = Serial.read();
        
        case 'F': // Avancer
            motor1.run(FORWARD);
            motor2.run(FORWARD);
        
        case 'B': // Reculer
            motor1.run(BACKWARD);
            motor2.run(BACKWARD);
        
        case 'L': // Gauche
            motor1.run(BACKWARD);
            motor2.run(FORWARD);
        
        case 'R': // Droite
            motor1.run(FORWARD);
            motor2.run(BACKWARD);
    }
}

// Serrure par Empreinte
#include 
#include 

SoftwareSerial fingerSerial(2, 3);
Adafruit_Fingerprint finger = Adafruit_Fingerprint(&fingerSerial);
Servo lockServo;

void setup() {
    Serial.begin(9600);
    finger.begin(57600);
    lockServo.attach(9);
    lockServo.write(0); // Verrouillé
}

void loop() {
    if(getFingerprintID() != -1) {
        lockServo.write(90); // Déverrouiller
        delay(5000);
        lockServo.write(0); // Verrouiller
    }
}
